def correlation_matrix(df, lbl_font_size=10):
    f = plt.figure(figsize=(25, 22))
    plt.matshow(df.corr(), fignum=f.number)
    plt.xticks(range(df.shape[1]), df.columns, fontsize=lbl_font_size, rotation=60)
    plt.yticks(range(df.shape[1]), df.columns, fontsize=lbl_font_size)
    cb = plt.colorbar()
    cb.ax.tick_params(labelsize=14, bottom=True, top=True, labelbottom=True, labeltop=True)
